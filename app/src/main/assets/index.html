Kode Arduino IDE ESP8266 (GAS HTTP)

#include <ESP8266WiFi.h> #include <ESP8266HTTPClient.h> #include <WiFiClientSecureBearSSL.h> #include <ArduinoJson.h> const char* ssid = "WIFI_SSID"; const char* password = "WIFI_PASSWORD"; // URL Google Apps Script Web App String GAS_URL = "https://script.google.com/macros/s/AKfycbyQctFYdJ77wJfE2gSMWjjfoa7FGpLTMzJ2bXf8e33Y3aJ7UGuN0PHNIE9Tn5_nLDl5/exec"; const int RELAY_PIN = D1; const int TRIG_PIN = D2; const int ECHO_PIN = D3; void setup_wifi() { WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); } } void sendDataAndGetCommands(int waterLevel) { if (WiFi.status() == WL_CONNECTED) { std::unique_ptr<BearSSL::WiFiClientSecure>client(new BearSSL::WiFiClientSecure); client->setInsecure(); // Abaikan sertifikat SSL untuk GAS HTTPClient https; // URL Request untuk update sensor dan get command String finalPath = GAS_URL + "?action=sync&waterLevel=" + String(waterLevel); if (https.begin(*client, finalPath)) { int httpCode = https.GET(); if (httpCode > 0 && httpCode == HTTP_CODE_OK) { String payload = https.getString(); // Parse JSON Respons: {"pumpStatus": true/false} DynamicJsonDocument doc(1024); deserializeJson(doc, payload); bool pumpStatus = doc["pumpStatus"]; if (pumpStatus) { digitalWrite(RELAY_PIN, HIGH); // NYALA } else { digitalWrite(RELAY_PIN, LOW); // MATI } } https.end(); } } } void setup() { pinMode(RELAY_PIN, OUTPUT); pinMode(TRIG_PIN, OUTPUT); pinMode(ECHO_PIN, INPUT); digitalWrite(RELAY_PIN, LOW); setup_wifi(); } void loop() { // Baca Sensor Ultrasonik digitalWrite(TRIG_PIN, LOW); delayMicroseconds(2); digitalWrite(TRIG_PIN, HIGH); delayMicroseconds(10); digitalWrite(TRIG_PIN, LOW); long duration = pulseIn(ECHO_PIN, HIGH); int distance = duration * 0.034 / 2; // Kalkulasi % air int percent = map(distance, 100, 10, 0, 100); percent = constrain(percent, 0, 100); // Sinkronisasi ke Google Sheet & Ambil Status Relay setiap 3 detik sendDataAndGetCommands(percent); delay(3000); }

Google Apps Script (Code.gs)

/* * 1. Buka Google Sheet Baru * 2. Klik Ekstensi > Apps Script * 3. Paste kode ini. Buat Sheet 'Dashboard' dan beri header: Action, Timestamp, dll * 4. Terapkan > Deployment Baru > Aplikasi Web */ function doGet(e) { var action = e.parameter.action; var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Dashboard"); // Jika parameter missing, berikan state standar if(!action) { return ContentService.createTextOutput(JSON.stringify({error: "No Action"})) .setMimeType(ContentService.MimeType.JSON); } // Menyimpan properti aplikasi menggunakan PropertiesService var props = PropertiesService.getScriptProperties(); if (action === "set") { // Android App mengirim push commands (e.g., pumpStatus=true) if(e.parameter.pumpStatus !== undefined) { props.setProperty("pumpStatus", e.parameter.pumpStatus); sheet.appendRow([new Date(), "PUMP_STATUS_CHANGED", e.parameter.pumpStatus]); } if(e.parameter.autoMode !== undefined) { props.setProperty("autoMode", e.parameter.autoMode); } return ContentService.createTextOutput(JSON.stringify({success: true})) .setMimeType(ContentService.MimeType.JSON); } else if (action === "sync") { // ESP8266 melaporkan water level dan meminta status if(e.parameter.waterLevel) { props.setProperty("waterLevel", e.parameter.waterLevel); sheet.appendRow([new Date(), "WATER_LEVEL", e.parameter.waterLevel]); } var currentPump = props.getProperty("pumpStatus") === "true"; return ContentService.createTextOutput(JSON.stringify({ pumpStatus: currentPump, status: "OK" })).setMimeType(ContentService.MimeType.JSON); } else if (action === "get") { // Android WebApp fetch semua state API var s = { pumpStatus: props.getProperty("pumpStatus") === "true", autoMode: props.getProperty("autoMode") === "true", waterLevel: parseInt(props.getProperty("waterLevel") || "0") }; return ContentService.createTextOutput(JSON.stringify(s)) .setMimeType(ContentService.MimeType.JSON); } }
